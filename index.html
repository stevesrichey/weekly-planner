<script>

/* ── State ── */

let A=[],P=[],S=null,WS=null;

const DN=['Monday','Tuesday','Wednesday','Thursday','Friday'];

const HH=[9,10,11,12,13,14,15,16];

const TGT=6;


/* ── Project Colors ── */

const PC=[

  {b:'#dbeafe',f:'#1e40af'},{b:'#dcfce7',f:'#166534'},{b:'#fce7f3',f:'#9d174d'},

  {b:'#fef3c7',f:'#92400e'},{b:'#e0e7ff',f:'#3730a3'},{b:'#ccfbf1',f:'#115e59'},

  {b:'#ffe4e6',f:'#9f1239'},{b:'#f3e8ff',f:'#6b21a8'},{b:'#fed7aa',f:'#9a3412'},

  {b:'#cffafe',f:'#155e75'},{b:'#fef9c3',f:'#854d0e'},{b:'#d1fae5',f:'#065f46'},

  {b:'#fbcfe8',f:'#831843'},{b:'#e2e8f0',f:'#334155'},{b:'#fee2e2',f:'#991b1b'}

];


/* ── Init ── */

document.addEventListener('DOMContentLoaded',()=>{

  load();setDate();rA();rP();

function setDate(){

  const d=new Date(),w=d.getDay();

  const add=w===0?1:w===1?0:8-w;

  d.setDate(d.getDate()+add);

  document.getElementById('ws').value=fdi(d);

}


/* ── Date helpers ── */

function fdi(d){return`${d.getFullYear()}-${p2(d.getMonth()+1)}-${p2(d.getDate())}`}

function pd(s){const[y,m,d]=s.split('-').map(Number);return new Date(y,m-1,d)}

function dpd(b,n){const d=new Date(b);d.setDate(d.getDate()+n);return d}

function fh(h){return h===12?'12:00 PM':h>12?`${h-12}:00 PM`:`${h}:00 AM`}

function fdd(d){return d.toLocaleDateString('en-US',{weekday:'long',month:'long',day:'numeric',year:'numeric'})}

function p2(n){return String(n).padStart(2,'0')}


/* ── Appointments ── */

function addA(){

  const day=+document.getElementById('aD').value;

  const hr=+document.getElementById('aT').value;

  const dur=+document.getElementById('aL').value;

  const nm=document.getElementById('aN').value.trim();

  if(!nm)return toast('Enter an appointment name');

  if(hr+dur>17)return toast('Extends past 5:00 PM');

  for(const a of A){

    if(a.day!==day)continue;

    if(hr<a.hr+a.dur&&a.hr<hr+dur)return toast(`Conflicts with "${a.nm}"`);

  }

  A.push({id:Date.now(),day,hr,dur,nm});

  document.getElementById('aN').value='';rA();save();

}

function rmA(id){A=A.filter(a=>a.id!==id);rA();save();}

function clearAppts(){if(A.length&&confirm('Clear all appointments?')){A=[];rA();save();}}

function rA(){

  const el=document.getElementById('aList');

  if(!A.length){el.innerHTML='<p class="emp">No appointments yet</p>';return}

  el.innerHTML=A.map(a=>`<div class="it"><div><span class="n">${esc(a.nm)}</span><span class="m">${DN[a.day]} ${fh(a.hr)}, ${a.dur}hr${a.dur>1?'s':''}</span></div><button class="br" onclick="rmA(${a.id})">✕</button></div>`).join('');

}


/* ── Projects ── */

function addP(){

  const nm=document.getElementById('pN').value.trim();

  if(!nm)return toast('Enter a project name');

  if(P.some(p=>p.nm.toLowerCase()===nm.toLowerCase()))return toast('Already listed');

  P.push({id:Date.now(),nm});

  document.getElementById('pN').value='';rP();save();

}

function rmP(id){P=P.filter(p=>p.id!==id);rP();save();}

function clearProjs(){if(P.length&&confirm('Clear all projects?')){P=[];rP();save();}}

function rP(){

  const el=document.getElementById('pList');

  if(!P.length){el.innerHTML='<p class="emp">No projects yet</p>';return}

  el.innerHTML=P.map((p,i)=>{

    const c=PC[i%PC.length];

    return`<div class="it"><span class="n" style="display:flex;align-items:center;gap:6px"><span class="ls" style="background:${c.b};border:1px solid ${c.f}40"></span>${esc(p.nm)}</span><button class="br" onclick="rmP(${p.id})">✕</button></div>`;

  }).join('');

}


/* ── Generate Schedule ── */

function gen(){

  const ws=document.getElementById('ws').value;

  if(!ws)return toast('Select a week start date');

  WS=pd(ws);

  if(WS.getDay()!==1)return toast('Please select a Monday');

  if(!P.length)return toast('Add at least one project');


  // Init grid

  S=Array.from({length:5},()=>{const d={};HH.forEach(h=>d[h]=null);return d});


  // 1. Place appointments

  for(const a of A)for(let h=a.hr;h<a.hr+a.dur&&h<=16;h++)S[a.day][h]={t:'a',nm:a.nm};


  // 2. Admin & R&D on Friday morning (fallback if taken)

  plF('d','Admin');plF('r','R&D');


  // 3. Collect project slots (fill each day to TGT)

  let ps=[];

  for(let d=0;d<5;d++){

    let filled=HH.filter(h=>S[d][h]).length;

    let need=Math.max(0,TGT-filled);

    for(const h of HH){if(need<=0)break;if(!S[d][h]){ps.push({d,h});need--;}}

  }


  // 4. Overflow check

  let warn=[];

  if(ps.length<P.length){

    let deficit=P.length-ps.length;

    for(let d=0;d<5&&deficit>0;d++){

      for(const h of HH){

        if(deficit<=0)break;

        if(!S[d][h]&&!ps.some(s=>s.d===d&&s.h===h)){ps.push({d,h});deficit--;}

      }

    }

    if(deficit>0){

      warn.push(`⚠️ Cannot schedule all ${P.length} projects even using buffer time. ${deficit} project(s) left unscheduled — consider trimming your list.`);

    }

  }


  // 5. Round-robin assign projects

  let pi=0;

  for(const s of ps){S[s.d][s.h]={t:'p',nm:P[pi].nm,pi};pi=(pi+1)%P.length;}


  // 6. Fill remaining with buffer

  for(let d=0;d<5;d++)HH.forEach(h=>{if(!S[d][h])S[d][h]={t:'b',nm:'Buffer'}});


  // Stats

  let st={p:0,a:0,d:0,r:0,b:0};

  for(let d=0;d<5;d++)HH.forEach(h=>{st[S[d][h].t]++});

  const sched=40-st.b;

  const pct=sched>0?Math.round(st.p/sched*100):0;


  // Buffer warning

  if(st.b<10)warn.push(`⚠️ Buffer reduced to ${st.b}hrs this week (ideal: 10hrs). This is exactly what this tool helps you avoid!`);


  // Appointments-heavy day check

  for(let d=0;d<5;d++){

    let apptH=HH.filter(h=>S[d][h].t==='a').length;

    if(apptH>=6)warn.push(`⚠️ ${DN[d]} has ${apptH} hours of appointments — no room for project work that day.`);

  }


  // Render

  rGrid();rLeg();rStats(st,pct);

  document.getElementById('wrA').innerHTML=warn.map(w=>`<div class="wb">${w}</div>`).join('');


  const out=document.getElementById('output');

  out.classList.remove('vis');

  void out.offsetWidth; // reflow

  out.style.display='block';

  out.classList.add('vis');

  setTimeout(()=>out.scrollIntoView({behavior:'smooth'}),100);

}


function plF(type,name){

  // Try Friday morning first

  for(const h of HH){if(!S[4][h]){S[4][h]={t:type,nm:name};return}}

  // Fallback: late-week afternoons

  for(let d=3;d>=0;d--)for(let h=16;h>=9;h--){if(!S[d][h]){S[d][h]={t:type,nm:name};return}}

}


/* ── Render Grid ── */

function rGrid(){

  let h='<div class="sg">';

  h+='<div class="gh" style="border-radius:8px 0 0 0">Time</div>';

  DN.forEach((d,i)=>h+=`<div class="gh"${i===4?' style="border-radius:0 8px 0 0"':''}>${d.slice(0,3)}</div>`);

  HH.forEach(hr=>{

    h+=`<div class="gt">${fh(hr)}</div>`;

    for(let d=0;d<5;d++){

      const c=S[d][hr];

      let cls='',sty='';

      if(c.t==='p'){const col=PC[c.pi%PC.length];sty=`background:${col.b};color:${col.f}`}

      else if(c.t==='a')cls='ca';else if(c.t==='d')cls='cd';else if(c.t==='r')cls='cr';else cls='cb';

      const txt=c.t==='b'?'—':c.nm;

      h+=`<div class="gc ${cls}" style="${sty}" title="${esc(c.nm)}">${esc(txt)}</div>`;

    }

  });

  h+='</div>';

  document.getElementById('grA').innerHTML=h;

}


function rLeg(){

  let h='';

  P.forEach((p,i)=>{const c=PC[i%PC.length];

    h+=`<div class="li"><div class="ls" style="background:${c.b};border:1px solid ${c.f}40"></div>${esc(p.nm)}</div>`;

  });

  h+=`<div class="li"><div class="ls" style="background:var(--acc-l)"></div>Appointment</div>`;

  h+=`<div class="li"><div class="ls" style="background:var(--pur-l)"></div>Admin</div>`;

  h+=`<div class="li"><div class="ls" style="background:var(--suc-l)"></div>R&D</div>`;

  h+=`<div class="li"><div class="ls" style="background:#f8fafc;border:1px solid var(--brd)"></div>Buffer</div>`;

  document.getElementById('lgA').innerHTML=`<div class="lg">${h}</div>`;

}


function rStats(st,pct){

  const pc=pct>=70?'#065f46':'#991b1b',pb=pct>=70?'var(--suc-l)':'var(--dan-l)';

  document.getElementById('stA').innerHTML=`<div class="sr">

    <div class="sb" style="background:var(--pri-l)"><div class="num" style="color:#1e40af">${st.p}h</div><div class="lbl">Project</div></div>

    <div class="sb" style="background:var(--acc-l)"><div class="num" style="color:#92400e">${st.a}h</div><div class="lbl">Appointments</div></div>

    <div class="sb" style="background:var(--pur-l)"><div class="num" style="color:#6d28d9">${st.d+st.r}h</div><div class="lbl">Admin + R&D</div></div>

    <div class="sb" style="background:var(--gry-l)"><div class="num">${st.b}h</div><div class="lbl">Buffer</div></div>

    <div class="sb" style="background:${pb}"><div class="num" style="color:${pc}">${pct}%</div><div class="lbl">Project Focus</div></div>

  </div>`;

}


/* ── Export: Notion ── */

function cpN(){

  if(!S)return;

  let t='# 📋 80/20 Weekly Plan\n\n';

  for(let d=0;d<5;d++){

    t+=`## ${fdd(dpd(WS,d))}\n\n`;

    for(const h of HH){

      const c=S[d][h];if(c.t==='b')continue;

      const icon=c.t==='p'?'🎯':c.t==='a'?'📌':c.t==='d'?'🔧':'🔬';

      t+=`- [ ] ${fh(h)} – ${fh(h+1)}: ${icon} ${c.nm}\n`;

    }

    t+='\n';

  }

  navigator.clipboard.writeText(t).then(

    ()=>toast('✓ Copied! Paste into Notion.'),

    ()=>{const ta=document.createElement('textarea');ta.value=t;document.body.appendChild(ta);ta.select();document.execCommand('copy');document.body.removeChild(ta);toast('✓ Copied!')}

  );

}


/* ── Export: ICS ── */

function dlI(){

  if(!S)return;

  const now=new Date();

  const stamp=`${now.getUTCFullYear()}${p2(now.getUTCMonth()+1)}${p2(now.getUTCDate())}T${p2(now.getUTCHours())}${p2(now.getUTCMinutes())}${p2(now.getUTCSeconds())}Z`;

  let ics='BEGIN:VCALENDAR\r\nVERSION:2.0\r\nPRODID:-//8020Planner//EN\r\nCALSCALE:GREGORIAN\r\nMETHOD:PUBLISH\r\nX-WR-CALNAME:80/20 Weekly Plan\r\n';


  for(let d=0;d<5;d++){

    const dt=dpd(WS,d);

    const ds=`${dt.getFullYear()}${p2(dt.getMonth()+1)}${p2(dt.getDate())}`;

    let h=9;

    while(h<=16){

      const c=S[d][h];

      if(c.t==='b'||c.t==='a'){h++;continue} // skip buffer & existing appointments

      let end=h+1;

      while(end<=16&&S[d][end]&&S[d][end].nm===c.nm&&S[d][end].t===c.t)end++;

      const uid=`8020-${ds}${p2(h)}-${Math.random().toString(36).substr(2,8)}@planner`;

      const cat=c.t==='p'?'Project Work':c.t==='d'?'Admin':'R&D';

      ics+=`BEGIN:VEVENT\r\nUID:${uid}\r\nDTSTAMP:${stamp}\r\nDTSTART:${ds}T${p2(h)}0000\r\nDTEND:${ds}T${p2(end)}0000\r\nSUMMARY:${iesc(c.nm)}\r\nDESCRIPTION:${iesc(cat+' — 80/20 Weekly Plan')}\r\nCATEGORIES:${cat}\r\nSTATUS:CONFIRMED\r\nTRANSP:OPAQUE\r\nEND:VEVENT\r\n`;

      h=end;

    }

  }

  ics+='END:VCALENDAR\r\n';


  const blob=new Blob([ics],{type:'text/calendar;charset=utf-8'});

  const a=document.createElement('a');

  a.href=URL.createObjectURL(blob);

  a.download=`8020-plan-${document.getElementById('ws').value}.ics`;

  a.click();URL.revokeObjectURL(a.href);

  toast('✓ Downloaded! Import into Outlook.');

}
